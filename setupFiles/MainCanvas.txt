package server.generated;

import processing.core.PApplet;
import server.CJamBlob;
import server.PStyleCopy;


public class MainCanvas extends PApplet {

	CJamBlob blobs[];
	
	@Override
	public void setup() {
		super.setup();
		size(1920, 1080);
		blobs = loadBlobs();
		for (CJamBlob b : blobs) {
			pushStyle();
			b.setup();
			b.style = PStyleCopy.copyStyle(g.getStyle());
			popStyle();
		}
	}


	@Override
	public void draw() {
		for (CJamBlob b : blobs) {
			pushStyle();
			style(b.style);
			b.draw();
			b.style = PStyleCopy.copyStyle(g.getStyle());
			popStyle();
		}
	}
